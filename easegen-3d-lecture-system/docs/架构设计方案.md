class CoursePlayer {
  constructor() {
    this.segments = [];
    this.currentIndex = 0;
    this.isPlaying = false;
    this.isPaused = false;
    this.xmovSDK = null;
  }

  async loadCourse(courseId) {
    // 1. è·å–æ‰€æœ‰ç‰‡æ®µ
    const response = await fetch(`/api/segments/${courseId}`);
    const data = await response.json();
    this.segments = data.data.segments;
    this.currentIndex = 0;
  }

  async play() {
    if (this.isPaused) {
      this.resume();
      return;
    }

    this.isPlaying = true;
    this.playSegment(this.currentIndex);
  }

  playSegment(index) {
    if (index >= this.segments.length) {
      this.onComplete();
      return;
    }

    const segment = this.segments[index];

    // 1. æ˜¾ç¤º PPT
    this.displayPPT(segment.pptImageUrl, index);

    // 2. é©±åŠ¨æ•°å­—äººè¯´è¯
    this.xmovSDK.speak(segment.text, true, true);

    // 3. ç›‘å¬è¯­éŸ³ç»“æŸï¼Œè‡ªåŠ¨æ’­æ”¾ä¸‹ä¸€æ®µ
    // (åœ¨ onVoiceStateChange å›è°ƒä¸­å¤„ç†)
  }

  onVoiceEnd() {
    if (this.isPlaying && !this.isPaused) {
      this.currentIndex++;
      this.playSegment(this.currentIndex);
    }
  }

  pause() {
    this.isPaused = true;
    this.xmovSDK.pause(); // å¦‚æœ SDK æ”¯æŒ
  }

  stop() {
    this.isPlaying = false;
    this.isPaused = false;
    this.currentIndex = 0;
    this.xmovSDK.stop(); // å¦‚æœ SDK æ”¯æŒ
  }

  next() {
    if (this.currentIndex < this.segments.length - 1) {
      this.currentIndex++;
      this.playSegment(this.currentIndex);
    }
  }

  previous() {
    if (this.currentIndex > 0) {
      this.currentIndex--;
      this.playSegment(this.currentIndex);
    }
  }
}
